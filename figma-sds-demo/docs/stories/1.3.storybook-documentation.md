### Recommended Story Structure

```typescript
// Individual variant stories
export const PricingCardDefault: StoryObj<typeof PricingCard> = {
  name: "Default Variant",
  args: {
    variant: "default",
    // ... other props
  },
};

// Playground story with full controls
export const PricingCardPlayground: StoryObj<typeof PricingCard> = {
  name: "Playground",
  args: {
    heading: "Pro Plan",
    price: "50",
    priceCurrency: "$",
    priceLabel: "/ mo",
    size: "large",
    variant: "stroke",
    action: "Get Started",
    actionVariant: "primary",
    list: ["Feature 1", "Feature 2", "Feature 3"],
  },
  argTypes: {
    size: { control: "select", options: ["large", "small"] },
    variant: { control: "select", options: ["default", "stroke", "brand"] },
    actionVariant: { control: "select", options: ["primary", "neutral", "subtle"] },
    actionDisabled: { control: "boolean" },
  },
};
```

### Coding Standards

- **Storybook Patterns:**
  - Use `StoryObj<typeof Component>` type
  - Use `args` for interactive props
  - Use `argTypes` for control configuration
  - Use `name:` for display name
  - Use `parameters.docs.description.story` for descriptions

[Source: architecture/10-coding-standards.md]
